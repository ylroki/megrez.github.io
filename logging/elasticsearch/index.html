Deploy to Kubernetes

apiVersion: v1
kind: ServiceAccount
metadata:
  name: elasticsearch-ske-database
  namespace: ske-database
  labels:
    k8s-app: elasticsearch-ske-database
    kubernetes.io/cluster-service: "true"
    addonmanager.kubernetes.io/mode: Reconcile
---
kind: ClusterRole
apiVersion: rbac.authorization.k8s.io/v1
metadata:
  name: elasticsearch-ske-database
  labels:
    k8s-app: elasticsearch-ske-database
    kubernetes.io/cluster-service: "true"
    addonmanager.kubernetes.io/mode: Reconcile
rules:
- apiGroups:
  - ""
  resources:
  - "services"
  - "namespaces"
  - "endpoints"
  verbs:
  - "get"
---
kind: ClusterRoleBinding
apiVersion: rbac.authorization.k8s.io/v1
metadata:
  namespace: ske-database
  name: elasticsearch-ske-database
  labels:
    k8s-app: elasticsearch-ske-database
    kubernetes.io/cluster-service: "true"
    addonmanager.kubernetes.io/mode: Reconcile
subjects:
- kind: ServiceAccount
  name: elasticsearch-ske-database
  namespace: ske-database
  apiGroup: ""
roleRef:
  kind: ClusterRole
  name: elasticsearch-ske-database
  apiGroup: ""
---
# Elasticsearch deployment itself
apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: elasticsearch-ske-database
  namespace: ske-database
  labels:
    k8s-app: elasticsearch-ske-database
    version: v7.1.0
    kubernetes.io/cluster-service: "true"
    addonmanager.kubernetes.io/mode: Reconcile
spec:
  serviceName: elasticsearch-ske-database
  replicas: 1
  selector:
    matchLabels:
      k8s-app: elasticsearch-ske-database
      version: v7.1.0
  template:
    metadata:
      labels:
        k8s-app: elasticsearch-ske-database
        version: v7.1.0
        kubernetes.io/cluster-service: "true"
    spec:
      serviceAccountName: elasticsearch-ske-database
      containers:
      - image: docker.elastic.co/elasticsearch/elasticsearch:7.1.0
        name: elasticsearch-ske-database
        resources:
          # need more cpu upon initialization, therefore burstable class
          limits:
            cpu: 8
          requests:
            cpu: 8
        ports:
        - containerPort: 9200
          name: db
          protocol: TCP
        - containerPort: 9300
          name: transport
          protocol: TCP
        volumeMounts:
        - name: elasticsearch-ske-database
          mountPath: /usr/share/elasticsearch/data
        env:
        - name: "NAMESPACE"
          valueFrom:
            fieldRef:
              fieldPath: metadata.namespace
        - name: "cluster.initial_master_nodes"
          value: "elasticsearch-ske-database-0"
      nodeName: devops-kube-test-node4
      volumes:
      - name: elasticsearch-ske-database
        hostPath: 
          path: /data/elasticsearch
          type: DirectoryOrCreate

      # Elasticsearch requires vm.max_map_count to be at least 262144.
      # If your OS already sets up this number to a higher value, feel free
      # to remove this init container.
      initContainers:
      - image: alpine:3.6
        command: ["/sbin/sysctl", "-w", "vm.max_map_count=262144"]
        name: elasticsearch-ske-database-init
        securityContext:
          privileged: true
---
apiVersion: v1
kind: Service
metadata:
  name: elasticsearch-ske-database
  namespace: ske-database
  labels:
    k8s-app: elasticsearch-ske-database
    kubernetes.io/cluster-service: "true"
    addonmanager.kubernetes.io/mode: Reconcile
    kubernetes.io/name: "Elasticsearch"
spec:
  ports:
  - port: 9200
    protocol: TCP
    targetPort: db
  selector:
    k8s-app: elasticsearch-ske-database

---
apiVersion: extensions/v1beta1
kind: Ingress
metadata:
  annotations:
    nginx.ingress.kubernetes.io/proxy-read-timeout: "3600"
    nginx.ingress.kubernetes.io/proxy-send-timeout: "3600"
    #nginx.ingress.kubernetes.io/auth-url:  
    nginx.ingress.kubernetes.io/rewrite-target: /$1
  name: elasticsearch-ske-database
  namespace: ske-database
spec:
  rules:
  - host: xxx
    http:
      paths:
      - backend:
          serviceName: elasticsearch-ske-database
          servicePort: 9200
        path: /elasticsearch/(.*)
  tls:
  - hosts:
    -  xxx
