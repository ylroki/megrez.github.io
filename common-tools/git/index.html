Git

https://git-scm.com/docs

Basic

# Initialize repository
$ git init
# Add modification
$ git add .
# Show Status
$ git status
# Commit
$ git commit .

Branch

# List branch
$ git branch -v
# Create branch
$ git branch xxx
# Delete branch
$ git branch -D xxx
# Working on branch
$ git checkout xxx
# Merge xxx into current branch
$ git merge xxx

Remote

# Clone remote repository
$ git clone ssh://aaa.git
# List remote hosts
$ git remote -v
# Add remote host
$ git remote add xxx ssh://aaa.git
# List branch include remote
$ git branch -a
# Checkout remote branch
$ git checkout xxx/aaa  -b aaa
# Pull
$ git pull 
# Push
$ git push
